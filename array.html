    #include<stdio.h>

        int isPallindrome(int num){
            int rev = 0, n = num;
            while (num != 0)
            {
                rev = (rev*10) + (num%10);
                num /= 10;
            }
            if (n == rev)
            {
                return 1;
            }
            return 0;
        }

    #include<stdio.h>

        int* merge(int arr1[], int s1, int arr2[], int s2, int arr[], int size){
            int i = 0, j = 0, k = 0;
            while (k < size)
            {
                if ((arr1[i] < arr2[j] || j == s2) && i != s1){
                    arr[k] = arr1[i];
                    k++;
                    i++;
                }
                else{
                    arr[k] = arr2[j];
                    k++;
                    j++;
                }
            }
            return arr;
        }
