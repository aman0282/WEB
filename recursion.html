CALCULATE FACTORIAL OF A NUMBER USING RECURSION

                        #include<stdio.h>
      
                            int factorial(int num)
                            {
                                if (num==0 || num ==1)
                                {
                                    return 1;
                                }
                                else
                                return num*factorial(num-1);
                            }
                            void main()
                            {
                                int num;
                                printf("Enter Number\n");
                                scanf("%d",&num);
                                printf("Factorial of %d is %d",num,factorial(num));
                            }

output

Enter Number
3
Factorial of 3 is 6

                            int findMax(int arr[], int n, int max) {

                                    return max;
                                }

FIND THE ARITHMETIC MEAN OF AN INTEGER ARRAY USING RECURSION.

                                #include<stdio.h>

                                    void getArray(int arr[],int s)
                                    {
                                        int i,j;
                                        printf("Enter Value of Array\n");
                                        for ( i = 0; i < s; i++)
                                        {
                                            scanf("%d",&arr[i]);
                                        }
                                        
                                    }
                                    double Mean(int arr[], int n, int index, double sum) {
                                        if (index == n) {
                                            return sum / n;
                                        }
                                        
                                        sum += arr[index];
                                        
                                        return Mean(arr, n, index + 1, sum);
                                    }
                                    
                                    int main() {
                                        int arr[10],s=10;
                                        getArray(arr,s);
                                        double mean = Mean(arr, s, 0, 0.0);
                                        printf("The arithmetic mean of the array is: %.2lf\n", mean);
                                    
                                        return 0;
                                    }
